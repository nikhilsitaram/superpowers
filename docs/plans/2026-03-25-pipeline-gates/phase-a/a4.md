# A4: Write check-base mode tests

**Avoid:** Don't actually change branches in the test — that would interfere with the working directory. Instead, mock `git rev-parse --abbrev-ref HEAD` by creating a wrapper script that the test controls. The simplest approach: create a temporary git repo in `$TMPDIR`, check out the desired branch name, and run `--check-base` from that directory. The `do_check_base()` function calls `git rev-parse --abbrev-ref HEAD` which will pick up the branch from `$TMPDIR`'s git context.

Actually, `do_check_base()` runs `git rev-parse --abbrev-ref HEAD` in the current working directory, not the plan directory. So we need to either:
1. `cd` to a temp git repo before running the command, or
2. Rely on the fact that the test runs inside a git repo (the main repo checkout) and test with a known branch.

The safest approach: initialize a git repo in `$TMPDIR` with a specific branch name, `cd` there, and run the validate-plan command. Use a subshell to avoid cwd pollution.

## Steps

### Step 1: Create test file with boilerplate

Create `tests/validate-plan/test_check_base.sh` with the standard test harness. Add a helper function to create a temporary git repo on a specific branch:

```bash
#!/usr/bin/env bash
set -euo pipefail

SCRIPT_DIR="$(cd "$(dirname "$0")" && pwd)"
REPO_ROOT="$(cd "$SCRIPT_DIR/../.." && pwd)"
VALIDATE="$REPO_ROOT/scripts/validate-plan"
FIXTURES="$SCRIPT_DIR/fixtures"
PASS=0
FAIL=0

assert_pass() {
  local desc="$1"; shift
  if "$@" > /dev/null 2>&1; then
    echo "PASS: $desc"
    ((PASS++)) || true
  else
    echo "FAIL: $desc"
    ((FAIL++)) || true
  fi
}

assert_fail() {
  local desc="$1"; shift
  local expected_error="$1"; shift
  local output
  if output=$("$@" 2>&1); then
    echo "FAIL: $desc (expected failure, got success)"
    ((FAIL++)) || true
  elif echo "$output" | grep -q "$expected_error"; then
    echo "PASS: $desc"
    ((PASS++)) || true
  else
    echo "FAIL: $desc (expected '$expected_error' in output, got: $output)"
    ((FAIL++)) || true
  fi
}

TMPDIR=$(mktemp -d)
trap 'rm -rf "$TMPDIR"' EXIT

setup_git_repo() {
  local branch_name="$1"
  local dir="$TMPDIR/repo"
  rm -rf "$dir"
  mkdir -p "$dir"
  git -C "$dir" init -q
  git -C "$dir" checkout -q -b "$branch_name" 2>/dev/null || git -C "$dir" switch -q -c "$branch_name"
  git -C "$dir" commit -q --allow-empty -m "init"
  echo "$dir"
}

setup_plan_in_repo() {
  local repo_dir="$1"
  local plan_dir="$repo_dir/docs/plans/test"
  mkdir -p "$plan_dir/phase-a"
  cp "$FIXTURES/valid-plan/plan.json" "$plan_dir/plan.json"
  cp -r "$FIXTURES/valid-plan/phase-a/"* "$plan_dir/phase-a/"
  mkdir -p "$plan_dir/phase-b"
  cp -r "$FIXTURES/valid-plan/phase-b/"* "$plan_dir/phase-b/"
  echo "$plan_dir"
}
```

### Step 2: Write single-phase tests (no integration_branch)

```bash
echo "Test 1: Single-phase plan on feature branch passes"
REPO_DIR=$(setup_git_repo "feat/pipeline-gates")
PLAN_DIR=$(setup_plan_in_repo "$REPO_DIR")
assert_pass "feature branch with no integration_branch passes" \
  bash -c "cd '$REPO_DIR' && '$VALIDATE' --check-base '$PLAN_DIR/plan.json'"

echo "Test 2: Single-phase plan on main fails"
REPO_DIR=$(setup_git_repo "main")
PLAN_DIR=$(setup_plan_in_repo "$REPO_DIR")
assert_fail "main branch with no integration_branch fails" "base_branch_mismatch" \
  bash -c "cd '$REPO_DIR' && '$VALIDATE' --check-base '$PLAN_DIR/plan.json'"

echo "Test 3: Single-phase plan on master fails"
REPO_DIR=$(setup_git_repo "master")
PLAN_DIR=$(setup_plan_in_repo "$REPO_DIR")
assert_fail "master branch with no integration_branch fails" "base_branch_mismatch" \
  bash -c "cd '$REPO_DIR' && '$VALIDATE' --check-base '$PLAN_DIR/plan.json'"
```

### Step 3: Write multi-phase tests (with integration_branch)

```bash
echo "Test 4: Multi-phase plan on matching integration branch passes"
REPO_DIR=$(setup_git_repo "integrate/my-feature")
PLAN_DIR=$(setup_plan_in_repo "$REPO_DIR")
jq '. + {"integration_branch": "integrate/my-feature"}' "$PLAN_DIR/plan.json" > "$PLAN_DIR/tmp.json" && mv "$PLAN_DIR/tmp.json" "$PLAN_DIR/plan.json"
assert_pass "matching integration branch passes" \
  bash -c "cd '$REPO_DIR' && '$VALIDATE' --check-base '$PLAN_DIR/plan.json'"

echo "Test 5: Multi-phase plan on wrong branch fails"
REPO_DIR=$(setup_git_repo "feat/wrong-branch")
PLAN_DIR=$(setup_plan_in_repo "$REPO_DIR")
jq '. + {"integration_branch": "integrate/my-feature"}' "$PLAN_DIR/plan.json" > "$PLAN_DIR/tmp.json" && mv "$PLAN_DIR/tmp.json" "$PLAN_DIR/plan.json"
assert_fail "wrong branch with integration_branch fails" "base_branch_mismatch" \
  bash -c "cd '$REPO_DIR' && '$VALIDATE' --check-base '$PLAN_DIR/plan.json'"

echo "Test 6: Multi-phase plan on main with integration_branch set fails"
REPO_DIR=$(setup_git_repo "main")
PLAN_DIR=$(setup_plan_in_repo "$REPO_DIR")
jq '. + {"integration_branch": "integrate/my-feature"}' "$PLAN_DIR/plan.json" > "$PLAN_DIR/tmp.json" && mv "$PLAN_DIR/tmp.json" "$PLAN_DIR/plan.json"
assert_fail "main branch with integration_branch set fails" "base_branch_mismatch" \
  bash -c "cd '$REPO_DIR' && '$VALIDATE' --check-base '$PLAN_DIR/plan.json'"
```
