# A2: Add file-set overlap detection to validate-plan

**Avoid:** Don't check file-set overlap across phases — only within the same phase. Cross-phase overlap is valid because phases execute sequentially. The design doc explicitly states this: "each task touches a unique file set — enforced at plan time" applies within a phase.

**Avoid:** Don't just check `create` arrays (that already exists). The new check covers all three arrays: `create`, `modify`, and `test`, and also detects cross-array overlap (e.g., task A1 creates `foo.ts` while task A2 modifies `foo.ts`).

## Steps

### Step 1: Write failing tests

Create `tests/validate-plan/test_fileset_overlap.sh`:

```bash
#!/usr/bin/env bash
set -euo pipefail

SCRIPT_DIR="$(cd "$(dirname "$0")" && pwd)"
VALIDATE="$SCRIPT_DIR/../../scripts/validate-plan"
TMPDIR_TEST=$(mktemp -d)
trap 'rm -rf "$TMPDIR_TEST"' EXIT

pass() { echo "PASS: $1"; }
fail() { echo "FAIL: $1"; exit 1; }

setup_phase() {
  local dir="$1"
  mkdir -p "$dir/phase-a" "$dir/phase-b" 2>/dev/null || true
  echo "" > "$dir/phase-a/completion.md"
  echo "" > "$dir/phase-b/completion.md"
}

make_task() {
  local id="$1" name="$2" create="$3" modify="$4" test_files="$5"
  cat <<TASK
{"id":"$id","name":"$name","status":"pending","depends_on":[],
 "files":{"create":[$create],"modify":[$modify],"test":[$test_files]},
 "verification":"echo ok","done_when":"Done"}
TASK
}

make_plan_two_tasks() {
  local dir="$1" t1_create="$2" t1_modify="$3" t1_test="$4" t2_create="$5" t2_modify="$6" t2_test="$7"
  setup_phase "$dir"
  echo "# A1: Task one" > "$dir/phase-a/a1.md"
  echo "# A2: Task two" > "$dir/phase-a/a2.md"
  local t1 t2
  t1=$(make_task "A1" "Task one" "$t1_create" "$t1_modify" "$t1_test")
  t2=$(make_task "A2" "Task two" "$t2_create" "$t2_modify" "$t2_test")
  cat > "$dir/plan.json" <<PLAN
{"schema":1,"status":"Not Yet Started","workflow":"create-pr",
 "goal":"Test","architecture":"Test","tech_stack":"Test",
 "phases":[{"letter":"A","name":"Phase A","status":"Not Started",
   "depends_on":[],"rationale":"Only phase",
   "tasks":[$t1,$t2]}]}
PLAN
}

D="$TMPDIR_TEST/t1"
make_plan_two_tasks "$D" "" "" "" "" "" ""
bash "$VALIDATE" --schema "$D/plan.json" && pass "no overlap passes" || fail "no overlap should pass"

D="$TMPDIR_TEST/t2"
make_plan_two_tasks "$D" '"src/foo.ts"' "" "" '"src/foo.ts"' "" ""
bash "$VALIDATE" --schema "$D/plan.json" 2>/dev/null && fail "same file in create of two tasks should fail" || pass "same create overlap detected"

D="$TMPDIR_TEST/t3"
make_plan_two_tasks "$D" "" '"src/bar.ts"' "" "" '"src/bar.ts"' ""
bash "$VALIDATE" --schema "$D/plan.json" 2>/dev/null && fail "same file in modify of two tasks should fail" || pass "same modify overlap detected"

D="$TMPDIR_TEST/t4"
make_plan_two_tasks "$D" "" "" '"tests/baz.test.ts"' "" "" '"tests/baz.test.ts"'
bash "$VALIDATE" --schema "$D/plan.json" 2>/dev/null && fail "same file in test of two tasks should fail" || pass "same test overlap detected"

D="$TMPDIR_TEST/t5"
make_plan_two_tasks "$D" '"src/qux.ts"' "" "" "" '"src/qux.ts"' ""
bash "$VALIDATE" --schema "$D/plan.json" 2>/dev/null && fail "create-modify cross-array overlap should fail" || pass "cross-array overlap detected"

D="$TMPDIR_TEST/t6"
mkdir -p "$TMPDIR_TEST/t6/phase-a" "$TMPDIR_TEST/t6/phase-b"
echo "" > "$TMPDIR_TEST/t6/phase-a/completion.md"
echo "" > "$TMPDIR_TEST/t6/phase-b/completion.md"
echo "# A1: Task one" > "$TMPDIR_TEST/t6/phase-a/a1.md"
echo "# B1: Task two" > "$TMPDIR_TEST/t6/phase-b/b1.md"
t1=$(make_task "A1" "Task one" '"src/shared.ts"' "" "")
t2=$(make_task "B1" "Task two" "" '"src/shared.ts"' "")
cat > "$TMPDIR_TEST/t6/plan.json" <<PLAN
{"schema":1,"status":"Not Yet Started","workflow":"create-pr",
 "goal":"Test","architecture":"Test","tech_stack":"Test",
 "phases":[
   {"letter":"A","name":"Phase A","status":"Not Started","depends_on":[],"rationale":"First",
    "tasks":[$t1]},
   {"letter":"B","name":"Phase B","status":"Not Started","depends_on":["A"],"rationale":"Second",
    "tasks":[$t2]}
 ]}
PLAN
bash "$VALIDATE" --schema "$TMPDIR_TEST/t6/plan.json" && pass "cross-phase overlap passes" || fail "cross-phase overlap should pass"

echo "All file-set overlap tests passed"
```

Run: `bash tests/validate-plan/test_fileset_overlap.sh` — tests 2-5 should fail (overlap not detected yet).

### Step 2: Implement file-set overlap detection in do_schema()

In `scripts/validate-plan`, inside the `do_schema()` function, after the per-task loop for each phase (after the `done` that closes the task loop, before the phase loop's `done`), add a within-phase file-set overlap check.

The approach: collect all file paths (from create, modify, test) for each task in the current phase into an associative array keyed by path, value is the first task ID that claimed it. If a second task claims the same path, emit an error.

```bash
    local -A phase_file_owners=()
    for ((t=0; t<task_count; t++)); do
      local tid
      tid=$(echo "$json" | jq -r ".phases[$p].tasks[$t].id // empty")
      for arr in create modify test; do
        local arr_count
        arr_count=$(echo "$json" | jq "(.phases[$p].tasks[$t].files.$arr // []) | length")
        for ((f=0; f<arr_count; f++)); do
          local fpath
          fpath=$(echo "$json" | jq -r ".phases[$p].tasks[$t].files.$arr[$f]")
          if [[ -n "${phase_file_owners[$fpath]+x}" ]]; then
            err "ERROR: fileset_overlap: file '$fpath' claimed by tasks ${phase_file_owners[$fpath]} and $tid in phase $letter"
          else
            phase_file_owners[$fpath]="$tid"
          fi
        done
      done
    done
```

Insert this block inside the phase loop, after the task iteration loop ends but before the phase loop's closing `done`.
