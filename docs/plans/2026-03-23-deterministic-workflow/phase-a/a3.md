# A3: Add --check-workflow flag to validate-plan

**Avoid:** Don't shell out to `gh pr list` without error handling. The gh CLI may not be authenticated, the repo may not have a remote, or the network may be down. Wrap gh calls in a function that captures stderr and provides a clear error message rather than letting raw gh errors propagate.

**Avoid:** Don't hardcode branch naming conventions. Read the integration branch name from git or derive it from plan metadata. The convention `integrate/<feature>` is used by the design skill, but the script should work with whatever branch names exist.

## Steps

### Step 1: Add the `do_check_workflow` function

Add after `do_check_review` in `scripts/validate-plan`. This function reads the `workflow` field from plan.json and runs gate checks based on its value.

```bash
do_check_workflow() {
  local plan_json="$1"
  local plan_dir
  plan_dir="$(cd "$(dirname "$plan_json")" && pwd)"

  local json
  json=$(cat "$plan_json")

  local workflow
  workflow=$(echo "$json" | jq -r '.workflow // empty')
  if [[ -z "$workflow" ]]; then
    echo "ERROR: missing workflow field in plan.json" >&2
    exit 1
  fi

  local gate_errors=()

  local dr_result
  dr_result=$(check_review_record "$plan_dir" "design-review" "design"); rc=$?
  if [[ $rc -ne 0 ]]; then
    gate_errors+=("design-review not passed: $dr_result")
  fi

  local pr_result
  pr_result=$(check_review_record "$plan_dir" "plan-review" "plan"); rc=$?
  if [[ $rc -ne 0 ]]; then
    gate_errors+=("plan-review not passed: $pr_result")
  fi

  if [[ "$workflow" == "plan-only" ]]; then
    if [[ ${#gate_errors[@]} -gt 0 ]]; then
      echo "ERROR: workflow gate (plan-only) unsatisfied:" >&2
      for ge in "${gate_errors[@]}"; do echo "  - $ge" >&2; done
      exit 1
    fi
    exit 0
  fi

  local plan_status
  plan_status=$(echo "$json" | jq -r '.status')
  if [[ "$plan_status" != "Complete" ]]; then
    gate_errors+=("plan status is '$plan_status', expected 'Complete'")
  fi

  local phase_count
  phase_count=$(echo "$json" | jq '.phases | length')
  for ((p=0; p<phase_count; p++)); do
    local letter
    letter=$(echo "$json" | jq -r ".phases[$p].letter")
    local letter_lower
    letter_lower=$(echo "$letter" | tr '[:upper:]' '[:lower:]')
    local ir_result
    ir_result=$(check_review_record "$plan_dir" "impl-review" "phase-$letter_lower"); rc=$?
    if [[ $rc -ne 0 ]]; then
      gate_errors+=("impl-review phase-$letter_lower: $ir_result")
    fi
  done

  if [[ "$phase_count" -gt 1 ]]; then
    local fr_result
    fr_result=$(check_review_record "$plan_dir" "impl-review" "final"); rc=$?
    if [[ $rc -ne 0 ]]; then
      gate_errors+=("impl-review final: $fr_result")
    fi
  fi

  # Nested function — bash nested functions share the enclosing scope,
  # so check_pr_exists can append to gate_errors directly.
  check_pr_exists() {
    local base="$1" head="$2" state="${3:-open}"
    local count
    count=$(gh pr list --base "$base" --head "$head" --state "$state" --json number --jq 'length' 2>/dev/null) || {
      gate_errors+=("gh pr list failed for base=$base head=$head — gh CLI may not be authenticated or network unavailable")
      return 1
    }
    if [[ "$count" -eq 0 ]]; then
      return 1
    fi
    return 0
  }

  local current_branch
  current_branch=$(git rev-parse --abbrev-ref HEAD 2>/dev/null || echo "unknown")

  if [[ "$current_branch" == integrate/* ]]; then
    local feature_name="${current_branch#integrate/}"
    for ((p=0; p<phase_count; p++)); do
      local letter
      letter=$(echo "$json" | jq -r ".phases[$p].letter")
      local letter_lower
      letter_lower=$(echo "$letter" | tr '[:upper:]' '[:lower:]')
      if ! check_pr_exists "$current_branch" "phase-$letter_lower" "all"; then
        gate_errors+=("no PR found for phase-$letter_lower → $current_branch")
      fi
    done
    if ! check_pr_exists "main" "$current_branch" "all"; then
      if ! check_pr_exists "master" "$current_branch" "all"; then
        gate_errors+=("no final PR found for $current_branch → main/master")
      fi
    fi
  else
    if ! check_pr_exists "main" "$current_branch" "all"; then
      if ! check_pr_exists "master" "$current_branch" "all"; then
        gate_errors+=("no PR found for $current_branch → main/master")
      fi
    fi
  fi

  if [[ "$workflow" == "merge-pr" ]]; then
    if [[ "$current_branch" == integrate/* ]]; then
      if ! check_pr_exists "main" "$current_branch" "merged"; then
        if ! check_pr_exists "master" "$current_branch" "merged"; then
          gate_errors+=("final PR not merged for $current_branch → main/master")
        fi
      fi
    else
      if ! check_pr_exists "main" "$current_branch" "merged"; then
        if ! check_pr_exists "master" "$current_branch" "merged"; then
          gate_errors+=("PR not merged for $current_branch → main/master")
        fi
      fi
    fi
  fi

  if [[ ${#gate_errors[@]} -gt 0 ]]; then
    echo "ERROR: workflow gate ($workflow) unsatisfied:" >&2
    for ge in "${gate_errors[@]}"; do echo "  - $ge" >&2; done
    exit 1
  fi
  exit 0
}
```

### Step 4: Write test file tests/validate-plan/test_check_workflow.sh

Create `tests/validate-plan/test_check_workflow.sh` following the existing test patterns. Test cases:

**plan-only workflow:**
1. **Plan-only passes with both reviews** -- design-review + plan-review passing, expect exit 0
2. **Plan-only fails missing design-review** -- only plan-review, expect exit 1
3. **Plan-only fails missing plan-review** -- only design-review, expect exit 1

**create-pr workflow:**
4. **Create-pr fails when plan not Complete** -- status "In Development" with all reviews, expect exit 1
5. **Create-pr fails missing impl-review** -- plan Complete but no impl-review for phase, expect exit 1
6. **Create-pr with all reviews but no PR** -- all reviews pass, plan Complete, but gh pr list returns 0 (skip this test if gh not available)

**merge-pr workflow:**
7. **Merge-pr requires PR merged** -- all reviews + PR exists but not merged state, expect exit 1

**Multi-phase specifics:**
8. **Multi-phase requires final impl-review** -- 2 phases, phase reviews pass, no `scope:final`, expect exit 1
9. **Single-phase skips final impl-review check** -- 1 phase, phase review passes, expect exit 0 (no final needed)

Note: PR verification tests depend on `gh` CLI availability. Guard with `command -v gh >/dev/null 2>&1 || { echo "SKIP: gh not available"; exit 0; }` or test only the review-record checks and skip PR checks.
